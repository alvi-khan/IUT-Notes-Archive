create table Course
(
Department varchar2(3),
Name varchar2(10),
Taught_By int,
Credit numeric (3, 2),

constraint Course_ID_Check primary key (Course_ID, Department),
constraint Course_to_Dept_Name_Check foreign key (Department) references Department (Dept_Name),
